82bool StandardGameLogic::checkLine(int row, int col, int dr, int dc, Player player) const {
83 int count = 0;
84 const int WinCondition = getWinCondition();
85
86 //Проверка в обе стороны от текущей позиции
87 for (int i = -WinCondition +1; i < WinCondition; i++) {
88 int r = row + i * dr;
89 int c = col + i * dc;
90
91 if (board_->getCell(r, c) == player) {
92 count++;
93 if (count >= WinCondition) {
94 return true;
95 }
96 } else {
97 count = 0;
98 }
99 }
100
101 return false;
102}
